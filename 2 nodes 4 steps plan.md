## Architecture Overview

> **QUICK RECAP**: A dual-node cognitive architecture with Logic and Symbolic processing connected by a Dynamic Bridge. Undergoes a 4-phase developmental curriculum to achieve system alignment.

The architecture consists of two primary processing nodes (Logic and Symbolic) connected by a Dynamic Bridge that routes information based on specific criteria. The system develops through a 4-phase cognitive alignment curriculum designed to build self-understanding, contextual awareness, empathy, and conceptual reasoning.

- Relies on traditional embedding approaches

> **QUICK RECAP**: Manages cluster-based meaning compression, tracks symbolic attractors, and facilitates recursive processing of abstract concepts.

**Key Characteristics:**
- Manages cluster-based meaning compression
- Tracks symbolic attractors (recurring patterns in symbol usage)
- Facilitates recursive processing of abstract concepts
- Uses symbolic clustering mechanisms
- Employs chaining for concept associations

- `symbol_cluster.py`
- `symbol_chainer.py`

- `symbol_suggester.py`

> **QUICK RECAP**: Routes information between nodes based on recursion depth, entropy measurements, and affective load.

- Routes information between Logic and Symbolic Nodes

```python
class ChainTracker:

        chain_id = hashlib.md5(chain_str.encode()).hexdigest()

        if chain_id in self.chains:

            self.chains[chain_id]['last_seen'] = self._get_timestamp()
            if metadata:

                'first_seen': self._get_timestamp(),
                'last_seen': self._get_timestamp(),
                'metadata': metadata or {}
            }
            
        # Update symbol to chain mapping
        for symbol in symbol_sequence:
            if symbol not in self.symbol_chains:
                self.symbol_chains[symbol] = set()
            self.symbol_chains[symbol].add(chain_id)
            
        # Update transition matrix
        for i in range(len(symbol_sequence) - 1):
            current = symbol_sequence[i]
            next_sym = symbol_sequence[i + 1]
            
            if current not in self.transition_matrix:
                self.transition_matrix[current] = {}
                
            if next_sym not in self.transition_matrix[current]:
                self.transition_matrix[current][next_sym] = 0
                
            self.transition_matrix[current][next_sym] += 1
            
        return chain_id
        
    def get_chain_strength(self, chain_id):
        """Calculate the strength of a chain based on frequency and recency."""
        if chain_id not in self.chains:
            return 0
            
        chain = self.chains[chain_id]
        
        # Base strength from frequency
        frequency_component = min(1.0, chain['frequency'] / 10.0)  # Cap at 10 occurrences
        
        # Recency component
        import datetime
        now = datetime.datetime.now()
        last_seen = chain['last_seen']
        days_since = (now - last_seen).days
        recency_component = max(0.0, 1.0 - (days_since / 30.0))  # Decay over 30 days
        
        # Combined strength score
        strength = 0.7 * frequency_component + 0.3 * recency_component
        
        return strength

        transitions = self.transition_matrix[symbol1]
        total_transitions = sum(transitions.values())
        
        if total_transitions == 0 or symbol2 not in transitions:
            return 0.0
            
        return transitions[symbol2] / total_transitions
```

        if len(symbolic_node.symbol_memory.clusters) <= limit:
            return 0
            
        # Calculate importance for each cluster
        cluster_importance = {}
        for name, cluster in symbolic_node.symbol_memory.clusters.items():
            # Base importance on:
            # 1. Number of symbols
            # 2. Average symbol importance
            # 3. Recency
            symbol_count = len(cluster['current_symbols'])
            
            # Average symbol importance
            importance_sum = 0
            for symbol in cluster['current_symbols']:
                importance_sum += symbolic_node.symbol_suggester.evaluate_symbol_importance(symbol)
                
            avg_importance = importance_sum / symbol_count if symbol_count > 0 else 0
            
            # Recency (most recent snapshot)
            import datetime
            now = datetime.datetime.now()
            if cluster['snapshots']:
                latest = max(snapshot['timestamp'] for snapshot in cluster['snapshots'])
                days_old = (now - latest).days
                recency_factor = max(0.1, 1.0 - (days_old / 30.0))  # Decay over 30 days
            else:
                recency_factor = 0.1
                
            # Combined importance
            cluster_importance[name] = (0.3 * symbol_count / 20.0 + 
                                      0.4 * avg_importance + 
                                      0.3 * recency_factor)
        
        # Sort clusters by importance
        sorted_clusters = sorted(
            cluster_importance.items(),
            key=lambda x: x[1],
            reverse=True
        )
        
        # Keep only the top clusters
        clusters_to_keep = set(name for name, _ in sorted_clusters[:limit])
        clusters_to_remove = []
        
        for name in symbolic_node.symbol_memory.clusters:
            if name not in clusters_to_keep:
                clusters_to_remove.append(name)
                
        # Remove excess clusters
        for name in clusters_to_remove:
            del symbolic_node.symbol_memory.clusters[name]
            
        return len(clusters_to_remove)
    
    # Step 3: Prune excess chains
    def prune_excess_chains(limit=max_chains):
        """Keep only the most important chains."""
        if len(symbolic_node.symbol_memory.chains) <= limit:
            return 0
            
        # Get strength for each chain
        chain_strength = {}
        for chain_id in symbolic_node.symbol_memory.chains:
            strength = symbolic_node.chain_tracker.get_chain_strength(chain_id)
            chain_strength[chain_id] = strength
            
        # Sort chains by strength
        sorted_chains = sorted(
            chain_strength.items(),
            key=lambda x: x[1],
            reverse=True
        )
        
        # Keep only the top chains
        chains_to_keep = set(chain_id for chain_id, _ in sorted_chains[:limit])
        chains_to_remove = []
        
        for chain_id in symbolic_node.symbol_memory.chains:
            if chain_id not in chains_to_keep:
                chains_to_remove.append(chain_id)
                
        # Remove excess chains
        for chain_id in chains_to_remove:
            del symbolic_node.symbol_memory.chains[chain_id]
            
        return len(chains_to_remove)
    
    # Apply all optimizations
    results = {
        'symbols_limit_function': limit_symbols_for_processing,
        'removed_clusters': prune_excess_clusters(),
        'removed_chains': prune_excess_chains()
    }
    
    return results

def _load_phase_specific_data(self, phase):
    """
    Load phase-specific reference data.
    
    Args:
        phase (int): The curriculum phase
        
    Returns:
        dict: Phase-specific data
    """
    # Define file paths for phase-specific data
    phase_data_files = {
        1: "data/phase1_identity_corpus.txt",
        2: "data/phase2_context_corpus.txt",
        3: "data/phase3_empathy_corpus.txt",
        4: "data/phase4_ambiguity_corpus.txt"
    }
    
    # Default data for each phase
    default_data = {
        1: {
            "concepts": ["system", "computation", "architecture", "processing", 
                        "algorithm", "function", "node", "memory"],
            "focus": "computational identity and self-understanding"
        },
        2: {
            "concepts": ["history", "science", "culture", "time", "space", 
                        "context", "reference", "environment"],
            "focus": "temporal and spatial references"
        },
        3: {
            "concepts": ["emotion", "feeling", "empathy", "self", "other", 
                        "psychology", "experience", "perspective"],
            "focus": "symbolic empathy and self/other distinctions"
        },
        4: {
            "concepts": ["ambiguity", "paradox", "recursion", "philosophy", 
                        "metaphysics", "concept", "abstraction", "uncertainty"],
            "focus": "conceptual ambiguity and recursive paradox"
        }
    }
    
    # Try to load phase data from file
    try:
        if phase in phase_data_files:
            file_path = phase_data_files[phase]
            
            if os.path.exists(file_path):
                with open(file_path, 'r') as f:
                    data = f.read()
                
                # Process the data (implementation dependent on format)
                processed_data = self._process_corpus_data(data, phase)
                
                # Merge with default data
                merged_data = default_data[phase].copy()
                merged_data.update(processed_data)
                
                return merged_data
    except Exception as e:
        print(f"Error loading phase {phase} data: {str(e)}")
    
    # Return default data if file loading fails
    return default_data.get(phase, {})

def _process_corpus_data(self, data, phase):
    """
    Process corpus data based on phase requirements.
    
    Args:
        data (str): Raw corpus data
        phase (int): The curriculum phase
        
    Returns:
        dict: Processed data
    """
    # Basic processing - extract concepts and examples
    lines = data.strip().split('\n')
    concepts = []
    examples = []
    
    for line in lines:
        line = line.strip()
        if line.startswith('#'):
            # Section header - ignore
            continue
        elif line.startswith('*'):
            # Concept
            concept = line[1:].strip()
            concepts.append(concept)
        elif line:
            # Example
            examples.append(line)
    
    # Phase-specific processing
    if phase == 1:
        # Identity phase - focus on system concepts
        return {
            "identity_concepts": concepts,
            "identity_examples": examples
        }
    elif phase == 2:
        # Context phase - focus on reference frames
        return {
            "context_concepts": concepts,
            "context_examples": examples
        }
    elif phase == 3:
        # Empathy phase - focus on emotional content
        emotion_examples = {}
        
        # Try to categorize examples by emotion
        for example in examples:
            # Simple emotion detection
            try:
                from emotion_handler import predict_emotions
                emotions = predict_emotions(example)
                if emotions and 'verified' in emotions:
                    for emotion, score in emotions['verified']:
                        if score > 0.5:  # Significant emotion
                            if emotion not in emotion_examples:
                                emotion_examples[emotion] = []
                            emotion_examples[emotion].append(example)
            except:
                pass
        
        return {
            "empathy_concepts": concepts,
            "emotion_examples": emotion_examples
        }
    elif phase == 4:
        # Ambiguity phase - focus on complex relationships
        # Analyze concept relationships
        concept_pairs = []
        for i, concept1 in enumerate(concepts):
            for concept2 in concepts[i+1:]:
                concept_pairs.append((concept1, concept2))
        
        return {
            "ambiguity_concepts": concepts,
            "concept_relationships": concept_pairs,
            "ambiguity_examples": examples
        }
    
    return {"concepts": concepts, "examples": examples}

```python
def calculate_entropy(self, data):
    """
    Calculate information entropy to measure uncertainty or randomness.
    Higher entropy indicates more uncertainty, which may require logical grounding.
    
    Args:
        data (dict): The data being processed
        
    Returns:
        float: Entropy score (0.0 to 1.0)
    """
    # Get the text content from data
    if isinstance(data, dict) and 'text' in data:
        text = data['text']
    elif isinstance(data, dict) and 'original_input' in data:
        text = data['original_input']
    elif isinstance(data, str):
        text = data
    else:
        return 0.5  # Default for unknown data format
    
    # Method 1: Vector dispersion
    # Calculate how dispersed the text embedding is from common patterns
    try:
        import numpy as np
        from vector_engine import encode_with_minilm
        
        # Get text embedding
        embedding = encode_with_minilm(text)
        
        # Calculate vector norm - higher norm might indicate more unusual content
        vector_norm = np.linalg.norm(embedding)
        
        # Normalize to 0-1 range (typical range might be 0.5-1.5)
        vector_dispersion = min(1.0, max(0.0, (vector_norm - 0.5) / 1.0))
        
        # Method 2: Linguistic complexity
        # Calculate lexical diversity (unique words / total words)
        words = text.lower().split()
        if len(words) > 0:
            lexical_diversity = len(set(words)) / len(words)
        else:
            lexical_diversity = 0.5
        
        # Method 3: Retrieval confidence
        # If we have retrieval results, check their confidence scores
        retrieval_uncertainty = 0.5  # Default
        if isinstance(data, dict) and 'logic_results' in data:
            logic_results = data['logic_results']
            if logic_results and isinstance(logic_results, list) and len(logic_results) > 0:
                # Lower similarity score = higher uncertainty
                if isinstance(logic_results[0], tuple) and len(logic_results[0]) > 0:
                    similarity = logic_results[0][0]
                    retrieval_uncertainty = 1.0 - similarity
        
        # Combine methods (weighted average)
        entropy = (0.4 * vector_dispersion + 
                  0.3 * lexical_diversity + 
                  0.3 * retrieval_uncertainty)
        
        return entropy
    
    except Exception as e:
        # Fall back to simple method if vector processing fails
        words = text.lower().split()
        if len(words) > 0:
            lexical_diversity = len(set(words)) / len(words)
            return min(1.0, lexical_diversity * 1.5)  # Scale up a bit
        else:
            return 0.5  # Default for empty text
```

```python
    """
```

        # Update progress, capped at 1.0

        """
        Calculate relevance of input to identity concepts (Phase 1).
        
        Args:
            text (str): Input text
            symbols (list): Extracted symbols
            
        Returns:
            float: Relevance score (0.0 to 1.0)
        """
        # Get identity concepts for Phase 1
        identity_concepts = self.phase_data[1].get("concepts", [])
        
        # Count concept occurrences in text
        text_lower = text.lower()
        concept_count = sum(1 for concept in identity_concepts if concept.lower() in text_lower)
        
        # Count concept occurrences in symbols
        symbol_names = [s.get('symbol', '').lower() for s in symbols]
        symbol_concept_count = sum(1 for concept in identity_concepts 
                                if any(concept.lower() in symbol for symbol in symbol_names))
        
        # Calculate relevance score
        if not identity_concepts:
            return 0.0
            
        text_relevance = min(1.0, concept_count / len(identity_concepts))
        symbol_relevance = min(1.0, symbol_concept_count / len(identity_concepts))
        
        # Combined relevance (weighted)
        relevance = 0.7 * text_relevance + 0.3 * symbol_relevance
        
        return relevance

        Calculate relevance of input to context concepts (Phase 2).

            float: Relevance score (0.0 to 1.0)

        context_concepts = self.phase_data[2].get("concepts", [])

        text_lower = text.lower()

                                if any(concept.lower() in symbol for symbol in symbol_names))

        text_relevance = min(1.0, concept_count / len(context_concepts))
        symbol_relevance = min(1.0, symbol_concept_count / len(context_concepts))
        
        # Combined relevance (weighted)
        relevance = 0.6 * text_relevance + 0.4 * symbol_relevance
        
        return relevance

        Calculate relevance of input to empathy concepts (Phase 3).
        
        Args:
            text (str): Input text
            emotions (list): Detected emotions
            symbols (list): Extracted symbols
            
        Returns:
            float: Relevance score (0.0 to 1.0)

        # Get empathy concepts for Phase 3
        empathy_concepts = self.phase_data[3].get("concepts", [])
        
        # Count concept occurrences
        text_lower = text.lower()
        concept_count = sum(1 for concept in empathy_concepts if concept.lower() in text_lower)
        
        # Count emotion intensity
        emotion_intensity = 0.0
        if emotions:
            if isinstance(emotions, list) and len(emotions) > 0 and isinstance(emotions[0], tuple):
                emotion_intensity = sum(score for _, score in emotions) / len(emotions)
            elif isinstance(emotions, dict):
                emotion_intensity = sum(emotions.values()) / len(emotions) if emotions else 0.0
        
        # Calculate relevance score
        if not empathy_concepts:
            return 0.0
            
        concept_relevance = min(1.0, concept_count / len(empathy_concepts))
        
        # Combined relevance (weighted)
        relevance = 0.5 * concept_relevance + 0.5 * emotion_intensity
        
        return relevance
    
    def _calculate_ambiguity_relevance(self, text, symbols, chains):
        """
        Calculate relevance of input to ambiguity concepts (Phase 4).
        
        Args:
            text (str): Input text
            symbols (list): Extracted symbols
            chains (list): Symbol chains
            
        Returns:
            float: Relevance score (0.0 to 1.0)
        """
        # Get ambiguity concepts for Phase 4
        ambiguity_concepts = self.phase_data[4].get("concepts", [])
        
        # Count concept occurrences
        text_lower = text.lower()
        concept_count = sum(1 for concept in ambiguity_concepts if concept.lower() in text_lower)
        
        # Calculate chain complexity
        chain_complexity = 0.0
        if chains and isinstance(chains, list):
            # Measure chain length and interconnections
            chain_length = sum(len(chain) for chain in chains) / max(1, len(chains))
            chain_complexity = min(1.0, chain_length / 5.0)  # Normalize to 0-1
        
        # Calculate relevance score
        if not ambiguity_concepts:
            return 0.0
            
        concept_relevance = min(1.0, concept_count / len(ambiguity_concepts))
        
        # Check for paradoxical or ambiguous phrases
        paradox_indicators = ["paradox", "contradiction", "both true and false", 
                            "ambiguous", "uncertain", "recursive", "self-reference"]
        paradox_count = sum(1 for indicator in paradox_indicators if indicator in text_lower)
        paradox_relevance = min(1.0, paradox_count / len(paradox_indicators))
        
        # Combined relevance (weighted)
        relevance = (0.4 * concept_relevance + 
                    0.3 * chain_complexity + 
                    0.3 * paradox_relevance)
        
        return relevance
    
    def _calculate_emotion_metrics(self):
        """
        Calculate metrics about emotion processing capability.
        
        Returns:
            dict: Emotion metrics
        """
        # Collect emotion data from symbolic node data
        all_emotions = set()
        emotion_intensities = {}
        
        for data in self.symbolic_node_data:
            if 'emotions' in data and data['emotions']:
                emotions = data['emotions']
                
                # Handle emotions in tuple format (emotion, score)
                if isinstance(emotions, list) and len(emotions) > 0 and isinstance(emotions[0], tuple):
                    for emotion, score in emotions:
                        all_emotions.add(emotion)
                        if emotion not in emotion_intensities:
                            emotion_intensities[emotion] = []
                        emotion_intensities[emotion].append(score)
                
                # Handle emotions in dictionary format
                elif isinstance(emotions, dict):
                    for emotion, score in emotions.items():
                        all_emotions.add(emotion)
                        if emotion not in emotion_intensities:
                            emotion_intensities[emotion] = []
                        emotion_intensities[emotion].append(score)
        
        # Calculate diversity (how many different emotions recognized)
        standard_emotions = {"joy", "sadness", "anger", "fear", "surprise", "disgust", "trust"}
        recognized_count = len(all_emotions.intersection(standard_emotions))
        diversity = recognized_count / len(standard_emotions) if standard_emotions else 0.0
        
        # Calculate depth (average max intensity across emotions)
        depth = 0.0
        if emotion_intensities:
            max_intensities = [max(intensities) if intensities else 0.0 
                             for intensities in emotion_intensities.values()]
            depth = sum(max_intensities) / len(max_intensities) if max_intensities else 0.0
        
        return {
            "diversity": diversity,
            "depth": depth,
            "recognized_emotions": list(all_emotions),
            "emotion_count": len(all_emotions)
        }

        }
        
        # Try to load phase data from file
        try:
            if phase in phase_data_files:
                file_path = phase_data_files[phase]
                
                if os.path.exists(file_path):
                    with open(file_path, 'r') as f:
                        data = f.read()
                    
                    # Process the data (implementation dependent on format)

                    # Merge with default data

                    merged_data.update(processed_data)

        Process corpus data based on phase requirements.
        
        Args:
            data (str): Raw corpus data
            phase (int): The curriculum phase
            
        Returns:
            dict: Processed data

        # Basic processing - extract concepts and examples
        lines = data.strip().split('\n')
        concepts = []
        examples = []
        
        for line in lines:
            line = line.strip()
            if line.startswith('#'):
                # Section header - ignore
                continue
            elif line.startswith('*'):
                # Concept
                concept = line[1:].strip()
                concepts.append(concept)
            elif line:
                # Example
                examples.append(line)
        
        # Phase-specific processing
        if phase == 1:
            # Identity phase - focus on system concepts
            return {
                "identity_concepts": concepts,
                "identity_examples": examples
            }
        elif phase == 2:

            return {
                "context_concepts": concepts,
                "context_examples": examples
            }
        elif phase == 3:
            # Empathy phase - focus on emotional content
            emotion_examples = {}
            
            # Try to categorize examples by emotion
            for example in examples:
                # Simple emotion detection
                try:
                    from emotion_handler import predict_emotions
                    emotions = predict_emotions(example)
                    if emotions and 'verified' in emotions:
                        for emotion, score in emotions['verified']:
                            if score > 0.5:  # Significant emotion
                                if emotion not in emotion_examples:
                                    emotion_examples[emotion] = []
                                emotion_examples[emotion].append(example)
                except:
                    pass
            
            return {
                "empathy_concepts": concepts,
                "emotion_examples": emotion_examples
            }
        elif phase == 4:
            # Ambiguity phase - focus on complex relationships
            # Analyze concept relationships
            concept_pairs = []
            for i, concept1 in enumerate(concepts):
                for concept2 in concepts[i+1:]:
                    concept_pairs.append((concept1, concept2))
            
            return {
                "ambiguity_concepts": concepts,
                "concept_relationships": concept_pairs,
                "ambiguity_examples": examples
            }
        
        return {"concepts": concepts, "examples": examples}

```python
        current_time = time.time()
        if current_time - self.last_save_time > self.save_interval:
            self.save_state()
            self.last_save_time = current_time

            "state_count": len(self.states)
```

```python
def _load_phase_specific_data(self, phase):
    """
    Load phase-specific reference data.
    
    Args:
        phase (int): The curriculum phase
        
    Returns:
        dict: Phase-specific data
    """
    # Define file paths for phase-specific data
    phase_data_files = {
        1: "data/phase1_identity_corpus.txt",
        2: "data/phase2_context_corpus.txt",
        3: "data/phase3_empathy_corpus.txt",
        4: "data/phase4_ambiguity_corpus.txt"
    }
    
    # Default data for each phase
    default_data = {
        1: {
            "concepts": ["system", "computation", "architecture", "processing", 
                        "algorithm", "function", "node", "memory"],
            "focus": "computational identity and self-understanding"
        },
        2: {
            "concepts": ["history", "science", "culture", "time", "space", 
                        "context", "reference", "environment"],
            "focus": "temporal and spatial references"
        },
        3: {
            "concepts": ["emotion", "feeling", "empathy", "self", "other", 
                        "psychology", "experience", "perspective"],
            "focus": "symbolic empathy and self/other distinctions"
        },
        4: {
            "concepts": ["ambiguity", "paradox", "recursion", "philosophy", 
                        "metaphysics", "concept", "abstraction", "uncertainty"],
            "focus": "conceptual ambiguity and recursive paradox"
        }
    }
    
    # Try to load phase data from file
    try:
        if phase in phase_data_files:
            file_path = phase_data_files[phase]
            
            if os.path.exists(file_path):
                with open(file_path, 'r') as f:
                    data = f.read()
                
                # Process the data (implementation dependent on format)
                processed_data = self._process_corpus_data(data, phase)
                
                # Merge with default data
                merged_data = default_data[phase].copy()
                merged_data.update(processed_data)
                
                return merged_data
    except Exception as e:
        print(f"Error loading phase {phase} data: {str(e)}")
    
    # Return default data if file loading fails
    return default_data.get(phase, {})

def _process_corpus_data(self, data, phase):
    """
    Process corpus data based on phase requirements.
    
    Args:
        data (str): Raw corpus data
        phase (int): The curriculum phase
        
    Returns:
        dict: Processed data
    """
    # Basic processing - extract concepts and examples
    lines = data.strip().split('\n')
    concepts = []
    examples = []
    
    for line in lines:
        line = line.strip()
        if line.startswith('#'):
            # Section header - ignore
            continue
        elif line.startswith('*'):
            # Concept
            concept = line[1:].strip()
            concepts.append(concept)
        elif line:
            # Example
            examples.append(line)
    
    # Phase-specific processing
    if phase == 1:
        # Identity phase - focus on system concepts
        return {
            "identity_concepts": concepts,
            "identity_examples": examples
        }
    elif phase == 2:
        # Context phase - focus on reference frames
        return {
            "context_concepts": concepts,
            "context_examples": examples
        }
    elif phase == 3:
        # Empathy phase - focus on emotional content
        emotion_examples = {}
        
        # Try to categorize examples by emotion
        for example in examples:
            # Simple emotion detection
            try:
                from emotion_handler import predict_emotions
                emotions = predict_emotions(example)
                if emotions and 'verified' in emotions:
                    for emotion, score in emotions['verified']:
                        if score > 0.5:  # Significant emotion
                            if emotion not in emotion_examples:
                                emotion_examples[emotion] = []
                            emotion_examples[emotion].append(example)
            except:
                pass
        
        return {
            "empathy_concepts": concepts,
            "emotion_examples": emotion_examples
        }
    elif phase == 4:
        # Ambiguity phase - focus on complex relationships
        # Analyze concept relationships
        concept_pairs = []
        for i, concept1 in enumerate(concepts):
            for concept2 in concepts[i+1:]:
                concept_pairs.append((concept1, concept2))
        
        return {
            "ambiguity_concepts": concepts,
            "concept_relationships": concept_pairs,
            "ambiguity_examples": examples
        }
    
    return {"concepts": concepts, "examples": examples}
```

## Dynamic Bridge Implementation

The Dynamic Bridge is responsible for routing information between the Logic and Symbolic nodes. It makes decisions based on three key metrics:

```python
def calculate_entropy(self, data):
    """
    Calculate information entropy to measure uncertainty or randomness.
    Higher entropy indicates more uncertainty, which may require logical grounding.
    
    Args:
        data (dict): The data being processed
        
    Returns:
        float: Entropy score (0.0 to 1.0)
    """
    # Get the text content from data
    if isinstance(data, dict) and 'text' in data:
        text = data['text']
    elif isinstance(data, dict) and 'original_input' in data:
        text = data['original_input']
    elif isinstance(data, str):
        text = data
    else:
        return 0.5  # Default for unknown data format
    
    # Method 1: Vector dispersion
    # Calculate how dispersed the text embedding is from common patterns
    try:
        import numpy as np
        from vector_engine import encode_with_minilm
        
        # Get text embedding
        embedding = encode_with_minilm(text)
        
        # Calculate vector norm - higher norm might indicate more unusual content
        vector_norm = np.linalg.norm(embedding)
        
        # Normalize to 0-1 range (typical range might be 0.5-1.5)
        vector_dispersion = min(1.0, max(0.0, (vector_norm - 0.5) / 1.0))
        
        # Method 2: Linguistic complexity
        # Calculate lexical diversity (unique words / total words)
        words = text.lower().split()
        if len(words) > 0:
            lexical_diversity = len(set(words)) / len(words)
        else:
            lexical_diversity = 0.5
        
        # Method 3: Retrieval confidence
        # If we have retrieval results, check their confidence scores
        retrieval_uncertainty = 0.5  # Default
        if isinstance(data, dict) and 'logic_results' in data:
            logic_results = data['logic_results']
            if logic_results and isinstance(logic_results, list) and len(logic_results) > 0:
                # Lower similarity score = higher uncertainty
                if isinstance(logic_results[0], tuple) and len(logic_results[0]) > 0:
                    similarity = logic_results[0][0]
                    retrieval_uncertainty = 1.0 - similarity
        
        # Combine methods (weighted average)
        entropy = (0.4 * vector_dispersion + 
                  0.3 * lexical_diversity + 
                  0.3 * retrieval_uncertainty)
        
        return entropy
    
    except Exception as e:
        # Fall back to simple method if vector processing fails
        words = text.lower().split()
        if len(words) > 0:
            lexical_diversity = len(set(words)) / len(words)
            return min(1.0, lexical_diversity * 1.5)  # Scale up a bit
        else:
            return 0.5  # Default for empty text
```

```python
    """
    Decide whether to route data to LogicNode or SymbolicNode.

        affective_load = self.calculate_affective_load(data)

        return "terminate"

    if entropy > 0.7:
        # High uncertainty requires logical grounding
        return "logic_node"

    if affective_load > 0.6:
        # High emotional content benefits from symbolic processing
        return "symbolic_node"

    try:
        # Try to access curriculum state if available
```

```python
def safe_process_input(curriculum_manager, text, emotions, max_retries=3):
    """
    Process input with error recovery.
    
    Args:
        curriculum_manager: CurriculumManager instance
        text (str): Input text
        emotions (list): Detected emotions
        max_retries (int): Maximum retry attempts
        
    Returns:
        str: Response from curriculum manager or error message
    """
    retries = 0
    
    while retries < max_retries:
        try:
            return curriculum_manager.process_input(text, emotions)
        except Exception as e:
            retries += 1
            print(f"Error processing input (attempt {retries}/{max_retries}): {str(e)}")
            
            # If this is the last retry, try a simplified processing approach
            if retries == max_retries:
                print("Attempting simplified processing...")
                try:
                    # Basic fallback processing
                    logic_results = curriculum_manager.logic_node.process_text_for_facts(text)
                    logic_response = curriculum_manager.logic_node.perform_inference(text, logic_results)
                    return f"[Fallback Response] {logic_response}"
                except:
                    return "[ERROR] Unable to process input after multiple attempts."
            
            # Wait briefly before retrying
            time.sleep(0.5)
    
    return "[ERROR] Maximum retry attempts exceeded."
```

```python
def process_with_dynamic_flow(text, emotions, logic_node, symbolic_node, dynamic_bridge):
    """
    Process text through the dynamic flow between Logic and Symbolic nodes.
    
    Args:
        text (str): Input text
        emotions (list): Detected emotions
        logic_node: Logic Node instance
        symbolic_node: Symbolic Node instance
        dynamic_bridge: Dynamic Bridge instance
        
    Returns:
        dict: Processed data and results
    """
    # Initial processing with both nodes
    retrieved_info = logic_node.process_text_for_facts(text)
    logic_inference = logic_node.perform_inference(text, retrieved_info)
    symbols = symbolic_node.process_text_for_symbols(text, emotions)
    
    # Prepare data for dynamic routing
    data = {
        'text': text,
        'emotions': emotions,
        'logic_results': retrieved_info,
        'logic_inference': logic_inference,
        'symbolic_results': symbols
    }
    
    # Dynamic routing through bridge
    recursion_depth = 0
    max_recursion = 5
    current_data = data
    
    while recursion_depth < max_recursion:
        # Determine route
        route = dynamic_bridge.determine_route(
            current_data, 
            recursion_depth=recursion_depth
        )
        
        # Check if routing is complete
        if route == "terminate":
            break
            
        # Transfer and process data
        current_data = dynamic_bridge.transfer_data(
            current_data, 
            "current", 
            route
        )
        
        # Increment recursion depth
        recursion_depth += 1
    
    # Return final processed data
    return current_data
```

```python
class SystemStateManager:
    """Manage system state persistence across sessions."""
    
    def __init__(self, state_file="system_state.json"):
        self.state_file = state_file
        
    def save_system_state(self, logic_node, symbolic_node, dynamic_bridge, curriculum_manager):
        """
        Save the state of system components.
        
        Args:
            logic_node: Logic Node instance
            symbolic_node: Symbolic Node instance
            dynamic_bridge: Dynamic Bridge instance
            curriculum_manager: Curriculum Manager instance
            
        Returns:
            bool: Success status
        """
        import json
        
        try:
            # Collect state from components
            state = {
                "timestamp": time.time(),
                "curriculum": {
                    "phase": curriculum_manager.phase,
                    "phase_progress": curriculum_manager.phase_progress,
                    "interaction_count": curriculum_manager.interaction_count
                },
                "dynamic_bridge": {
                    "max_recursion_depth": dynamic_bridge.max_recursion_depth,
                    "routes_taken": dynamic_bridge.routes_taken[-100:] if hasattr(dynamic_bridge, "routes_taken") else []
                },
                "logic_node": {
                    "name": logic_node.name
                    # Add more logic node state
                },
                "symbolic_node": {
                    "name": symbolic_node.name,
                    "attractors": getattr(symbolic_node, "attractors", {})
                    # Add more symbolic node state
                }
            }
            
            # Save to file
            with open(self.state_file, 'w') as f:
                json.dump(state, f)
                
            print(f"System state saved to {self.state_file}")
            return True
            
        except Exception as e:
            print(f"Error saving system state: {str(e)}")
            return False
    
    def load_system_state(self, logic_node, symbolic_node, dynamic_bridge, curriculum_manager):
        """
        Load system state and apply to components.
        
        Args:
            logic_node: Logic Node instance
            symbolic_node: Symbolic Node instance
            dynamic_bridge: Dynamic Bridge instance
            curriculum_manager: Curriculum Manager instance
            
        Returns:
            bool: Success status
        """
        import json
        import os
        
        if not os.path.exists(self.state_file):
            print(f"State file {self.state_file} not found.")
            return False
            
        try:
            # Load state from file
            with open(self.state_file, 'r') as f:
                state = json.load(f)
                
            # Apply state to components
            
            # Curriculum manager
            if "curriculum" in state:
                curriculum = state["curriculum"]
                curriculum_manager.phase = curriculum.get("phase", 1)
                curriculum_manager.phase_progress = curriculum.get("phase_progress", {1: 0.0})
                curriculum_manager.interaction_count = curriculum.get("interaction_count", 0)
            
            # Dynamic bridge
            if "dynamic_bridge" in state:
                bridge = state["dynamic_bridge"]
                dynamic_bridge.max_recursion_depth = bridge.get("max_recursion_depth", 5)
                if hasattr(dynamic_bridge, "routes_taken") and "routes_taken" in bridge:
                    dynamic_bridge.routes_taken = bridge["routes_taken"]
            
            # Symbolic node
            if "symbolic_node" in state and "attractors" in state["symbolic_node"]:
                symbolic_node.attractors = state["symbolic_node"]["attractors"]
            
            print(f"System state loaded from {self.state_file}")
            return True
            
        except Exception as e:
            print(f"Error loading system state: {str(e)}")
            return False
```

            "logic_node": self._capture_logic_node_state(logic_node),
